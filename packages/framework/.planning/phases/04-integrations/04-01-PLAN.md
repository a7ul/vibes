<!-- Verified APIs from 04-RESEARCH.md. Use these exactly - do not reference existing reference/integrations/ docs. -->

MCPStdioClient:
  constructor: new MCPStdioClient({ command, args?, env? }, options?)
  methods: connect(), disconnect(), listTools(), callTool(name, args), getServerInstructions()

MCPHttpClient:
  constructor: new MCPHttpClient({ url, headers? }, options?)
  methods: same as MCPStdioClient

MCPToolset:
  constructor: new MCPToolset(client, options?)
  options: { toolCacheTtlMs?: number, instructions?: boolean }
  methods: tools(ctx), getServerInstructions(), invalidateCache()

MCPManager:
  constructor: new MCPManager()   <!-- NO arguments -->
  methods: addServer(client, opts?), connect(), disconnect(), tools(ctx), getServerInstructions()
  property: serverCount
  note: MCPManager itself implements Toolset - pass it directly to agent

createManagerFromConfig(path):
  returns: Promise<MCPManager> (connected)

AGUIAdapterOptions:
  deps?: TDeps                    <!-- NOT depsFactory -->
  getState?: () => Record | Promise<Record>

AGUIAdapter:
  constructor: new AGUIAdapter(agent, options?)
  handleRequest(input: AGUIRunInput): Response   <!-- takes AGUIRunInput, NOT Request -->
  handler(): (req: Request) => Promise<Response>  <!-- use this for Deno HTTP servers -->

AGUIRunInput:
  { threadId: string, runId?: string, messages: AGUIMessage[], state?: Record }
</interfaces>
